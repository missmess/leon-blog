# 前端# Vue2# 架构# 造轮子

去年抽空花了半天时间，手敲了个Vue的简易版，能够实现Vue的常规使用、响应式、虚拟dom、组件化等。但是布局渲染部分，是使用的虚拟dom转换为真实dom，然后全量插入到Vue的根节点。每次刷新都是全量，这个性能可见一斑。

这次花点时间，在原来版本的基础上，尝试加上VNode的diff算法，然后在diff过程中差量更新真实dom树，实现一个真正能用的vue版本。

声明！本次依然不是vue源码解析，是我自己基于vue的diff算法的理解，在原来的自制vue版本基础上加进去的。不代表官方实现！

源码依然贴在最后面。

节点key

在前面，我们实现了通过渲染函数生成虚拟dom树（VNode）。diff算法就是为了对比出新的Vnode树和旧的VNode树的差异：哪些VNode是新增的节点，哪些是要删除的节点，哪些是移动了位置的节点。对于新增的节点，我们要创建对应的真实dom；对于删除的节点，我们要删除对应的dom；对于移动位置的节点，我们可以复用原来的dom，只需要更新dom属性即可。

在vue中，比对是根据节点的key和节点类型来判断的。如果key和类型一致，则视为同一节点；不一致则为不同节点。每个节点都有一个名为key的属性。如果未提供key，默认使用列表中的index。提供了key的话，比较起来会更高效。

先加上创建key的函数：

/**
 * 创建VNode的key
 * 默认key为type-idx
 * 如果提供了props.key，则为type-key
 */
function _createKey(
  type: string,
  key: string | undefined,
  idx: number = 0
): string {
  return key ? `${type}-${key}` : `${type}-${idx}`;
}

然后改造一下h函数，让它在生成VNode的同时生成key。

/** h函数帮助生成虚拟节点VNode */
function h(type: string, props: any, children: (VNode | string | null)[] = []): VNode {
  // 存储为VNode
  let vnode: VNode = { type, props, children: [], key: _createKey(type, props.key) };
  let nonNullChildren = children.filter((child) => child !== null);
  // 添加children
  for (let idx = 0; idx < nonNullChildren.length; idx++) {
    const child = nonNullChildren[idx];
    if (typeof child === "string") {
      vnode.children.push({
        type: "text",
        props: { textContent: child },
        children: [],
        key: _createKey("text", undefined, idx),
      });
    } else {
      // 重新计算key
      child.key = _createKey(child.type, child.props.key, idx);
      vnode.children.push(child);
    }
  }

  return vnode;
}

这里没有考虑静态节点（不根据状态变化的节点）优化，所以节点如果不提供key的话，会跟它所处的index有关。那么只要发生中间元素的插入或删除，都会导致后面元素的key发生变化，匹配不上导致性能下降。

双端diff算法

有了key之后，就可以根据key来判断新旧两组VNode，是如何变化的。vue采用了一种很高效的双端diff算法，即同时从新旧VNode数组的前后两端共4个指针，尝试相互匹配并向中间移动：即判断新前.key==旧前.key、新后.key==旧后.key、新前.key==旧后.key、新后.key==旧前.key，匹配到了则指针往中间移动一位。这四种匹配已经满足了大部分的列表变更情形：比如在列表中插入一个或多个元素，或者删除一个或多个元素。只要不是位置互换，这四种匹配都一定能匹配上。但是不保险，假如发生了位置很错乱的互换，那么这四种就可能匹配不到。所以还得有一个backup方案，就是直接在旧数组里面，查找key==新前.key的，如果查不到，那证明该元素是新增的，查到了则将旧数组这个位置的VNode置空，防止被再次匹配上。这种性能会差一些，但一般走不到。

如果最后有一方指针走完了（新前指针大于新后指针或者旧前指针大于旧后指针），则代表新旧数组不可能再有匹配的了。如果这时候新数组还有剩余，那都是要新建的；如果旧数组还有剩余，那都是要删除的。

在遍历过程中，要实时更新真实dom，比如创建、删除dom元素、或者只需要更新元素属性。有一些特殊情况，就是创建的要考虑dom元素插入的位置。比如旧前.key==新后.key，那就需要将旧前对应的dom移动到旧后对应的dom后面。同理如果某VNode.key==新前.key，那么需要将该节点的dom移动到旧前对应的dom前面（记住始终变动的是旧列表，所以位置都是以旧列表的指针来的）。

完整的diff算法如下：

/**
 * 遍历以及diff一组新旧VNode，同时对parent下的真实dom进行创建、更新、移动或删除。
 * 从而以最小的变动让真实dom和最新VNode树保持一致。
 */
_renderDomWithDiff(oldChildren: VNode[], newChildren: VNode[], parent: HTMLElement) {
  let oldStartIdx = 0;
  let oldEndIdx = oldChildren.length - 1;
  let newStartIdx = 0;
  let newEndIdx = newChildren.length - 1;

  while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
    // 旧节点可能已被匹配，被置空了，跳过
    if (!oldChildren[oldStartIdx]) {
      oldStartIdx++;
    } else if (!oldChildren[oldEndIdx]) {
      oldEndIdx--;
    } else if (
      oldChildren[oldStartIdx].key === newChildren[newStartIdx].key
    ) {
      // 旧开始节点和新开始节点key相同，无需移动dom
      console.log(`【${this.name}】[${oldChildren[oldStartIdx].key}]匹配到节点`);
      this._patch(oldChildren[oldStartIdx], newChildren[newStartIdx], parent);
      oldStartIdx++;
      newStartIdx++;
    } else if (oldChildren[oldEndIdx].key === newChildren[newEndIdx].key) {
      // 旧结束节点和新结束节点key相同，无需移动dom
      console.log(`【${this.name}】[${oldChildren[oldEndIdx].key }]匹配到节点`);
      this._patch(oldChildren[oldEndIdx], newChildren[newEndIdx], parent);
      oldEndIdx--;
      newEndIdx--;
    } else if (oldChildren[oldStartIdx].key === newChildren[newEndIdx].key) {
      // 旧开始节点和新结束节点key相同，需要移动到旧结束节点后面
      console.log(`【${this.name}】[${oldChildren[oldStartIdx].key}]匹配到节点`);
      const node = this._patch(
        oldChildren[oldStartIdx],
        newChildren[newEndIdx],
        parent
      );
      parent.insertBefore(node, oldChildren[oldEndIdx]._domNode!.nextSibling);
      oldStartIdx++;
      newEndIdx--;
    } else if (oldChildren[oldEndIdx].key === newChildren[newStartIdx].key) {
      // 旧结束节点和新开始节点key相同，需要移动到旧开始节点前面
      console.log(`【${this.name}】[${oldChildren[oldEndIdx].key}]匹配到节点`);
      const node = this._patch(
        oldChildren[oldEndIdx],
        newChildren[newStartIdx],
        parent
      );
      parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
      oldEndIdx--;
      newStartIdx++;
    } else {
      // 前面四种情况都不匹配，尝试从旧数组中匹配新开始节点
      const idxInOld = oldChildren.findIndex(
        (node) => node && node.key === newChildren[newStartIdx].key
      );
      if (idxInOld !== -1) {
        // 找到了，需要移动到旧开始节点前面
        console.log(`【${this.name}】[${newChildren[newStartIdx].key}]匹配到节点`);
        const node = this._patch(
          oldChildren[idxInOld],
          newChildren[newStartIdx],
          parent
        );
        parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
        // 置空，防止反复鞭尸
        // @ts-ignore
        oldChildren[idxInOld] = undefined;
      } else {
        // 找不到，创建dom，插入到旧开始节点前面
        console.log(`【${this.name}】[${newChildren[newStartIdx].key}]是新增节点`);
        // 插入到oldStartIdx前面正好保证顺序
        const node = this._patch(undefined, newChildren[newStartIdx], parent);
        parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
      }
      newStartIdx++;
    }
  }

  // 新节点比较完毕，旧节点还有剩余
  if (oldStartIdx <= oldEndIdx) {
    for (let i = oldStartIdx; i <= oldEndIdx; i++) {
      if (!oldChildren[i]) continue;
      // 删除掉对应的dom
      console.log(`【${this.name}】[${oldChildren[i].key}]是删除节点`);
      parent.removeChild(oldChildren[i]._domNode!);
      // 如果是组件，需要销毁
      if (this.components.hasOwnProperty(oldChildren[i].type)) {
        this._unmountComponent(oldChildren[i].key);
      }
    }
  } else if (newStartIdx <= newEndIdx) {
    // 旧节点比较完毕，新节点还有剩余
    for (let i = newStartIdx; i <= newEndIdx; i++) {
      console.log(`【${this.name}】[${newChildren[i].key}]是新增节点`);
      // 创建dom，插入到parent末尾
      const node = this._patch(undefined, newChildren[i], parent);
      parent.insertBefore(node, newChildren[newEndIdx + 1]?._domNode || null);
    }
  }
}

这样，只需要告诉新旧两个VNode树，和它们的parent节点。调用该方法后，就能在parent节点下，差量更新它的children dom元素。

_patch方法

里面还有一个重要的方法，就是_patch方法。这个方法就是处理那些key相同的VNode，如果是text，则更新文本内容。如果是组件，则复用组件更新props。否则为dom，让它们更新属性。（额外的情况，如果旧VNode为空，则创建对应的dom）。

同时需要在这里处理递归，调用_renderDomWithDiff函数。

/**
 * 创建或更新真实dom节点。
 * 1. 如果有oldVNode：复用dom，只更新dom属性。
 * 2. 如果没有oldVNode：创建新的dom
 * 3. 递归处理children
 */
_patch(oldVNode: VNode | undefined, vnode: VNode, parent: HTMLElement): Node {
  // 1. 文本节点
  if (vnode.type === "text") {
    if (!oldVNode) {
      // 没有则创建
      const node = document.createTextNode(vnode.props.textContent);
      vnode._domNode = node;
    } else {
      if (oldVNode.props.textContent !== vnode.props.textContent) {
        oldVNode._domNode!.textContent = vnode.props.textContent;
      }
      vnode._domNode = oldVNode._domNode;
    }
    return vnode._domNode!;
  }

  // 2. 组件节点
  if (this.components.hasOwnProperty(vnode.type)) {
    if (!oldVNode) {
      // 没有则创建
      let component = this._componentInstances[vnode.key];
      if (!component) {
        component = new Vue(this.components[vnode.type]);
        component._innerSetName(vnode.type);
        this._componentInstances[vnode.key] = component;
      }
      // 更新proxy的props
      for (const key in vnode.props) {
        component.proxy[key] = vnode.props[key];
      }
      vnode._component = component;
    } else {
      // 否则只需更新
      for (const key in vnode.props) {
        oldVNode._component!.proxy[key] = vnode.props[key];
      }
      vnode._component = oldVNode._component;
    }

    vnode._component!.el = parent;
    // 子组件去渲染自己的元素即可
    vnode._component!._doRender();
    vnode._domNode = vnode._component!._lastVNode!._domNode!;
    return vnode._domNode;
  }

  // 3. 其他节点
  if (!oldVNode) {
    // 没有则创建
    const element = document.createElement(vnode.type);
    // 设置属性
    _updateProps(element, vnode.props);
    vnode._domNode = element;
  } else {
    // 否则只需更新
    _updateProps(
      oldVNode._domNode as HTMLElement,
      vnode.props,
      oldVNode.props
    );
    vnode._domNode = oldVNode._domNode;
  }

  // 递归处理children
  this._renderDomWithDiff(
    oldVNode?.children || [],
    vnode.children,
    vnode._domNode as HTMLElement
  );

  return vnode._domNode!;
}

到这里通过双端diff算法差量更新真实dom的工作就全部做完了。现在我们需要将递归渲染函数，由原来的全量创建版本_renderRecursively改为现在写的差量更新版本_renderDomWithDiff。

区分App组件和子组件

现在每个Vue组件都可以使用_renderDomWithDiff函数，来将自己下面的元素都绘制到this.el上了。但是子组件还好，它能始终保证它的this.el下面children都是由我们的渲染机制渲染出来的。而App组件就无法保证了，因为它挂载在一个根dom元素上，可能这个根dom元素上本来就有其他元素。这样我们差量更新可能会导致更新了错误的元素。所以我们要区分一下App根组件和其他组件。

参考Vue3，它是通过createApp方法来创建一个App级别的Vue实例的。我们也一样：

/** 创建一个根Vue组件 */
export function createApp(options: VueOptions) {
  const vue = new Vue(options);
  vue._isAppRoot = true;
  vue._innerSetName("App");
  // @ts-ignore
  window.__vue_app__ = vue;
  return vue;
}

然后我们在执行_renderDomWithDiff函数之前，如果首次渲染，需要清空元素。

// 执行渲染
_doRender() {
  if (this.el) {
    const vnodeTree = this.render(h.bind(this));
    // 根组件首次渲染，清空el下所有元素，防止VNode和真实dom不一致
    if (this._isAppRoot) {
      // 首次渲染
      if (!this._lastVNode) {
        this.el.innerHTML = "";
      }
    }

    // 使用diff算法重新渲染子树
    this._renderDomWithDiff(
      this._lastVNode ? [this._lastVNode] : [],
      [vnodeTree],
      this.el
    );
    // 记录下来
    this._lastVNode = vnodeTree;
  }
}

异步更新

整个差量更新渲染流程已经ready。但是还有一个问题，现在是同步执行的渲染，初始化或者每次数据变更，都会导致一个_doRender()差量更新。虽然已经差量更新了，但也架不住可能出现的1秒更新100次吧...

所以参考vue，我们加一下异步更新，即多次发起render，但是一段时间只会执行一次实际渲染。

我们通过微任务+防抖实现，这样的话，同步代码触发的不管多少次，都会在同步代码执行结束后，只执行一次实际渲染。

// 异步更新队列，在微任务结束后执行渲染
_queueRender() {
  // 防抖
  if (!this._updateQueued) {
    this._updateQueued = true;
    Promise.resolve().then(() => {
      this._doRender();
      this._updateQueued = false;
    });
  }
}

初始化和proxy状态变更触发，都调用_queueRender：

  _init() {

其他

其他还有一些变动，诸如组件卸载逻辑、dom属性更新逻辑。都不是重点，不再介绍，简单看下源码即可。

源码

加上了diff算法差量更新，以及异步更新后，最新的手搓简版vue代码如下：

/** Vue设置选项（同Vue2） */
export type VueOptions = {
  name?: string;
  el?: HTMLElement | null;
  components?: Record<string, VueOptions>;
  props?: string[];
  data?: () => Record<string, any>;
  watch?: Record<string, (newVal: any, oldVal: any) => void>;
  methods?: Record<string, any>;
  mounted?: () => void;
  destroyed?: () => void;
  render: (
    h: (tag: string, opts: any, children?: (VNode | string | null)[]) => VNode
  ) => VNode;

  [key: string]: any;
};

/** 虚拟node节点 */
export type VNode = {
  key: string;
  type: string;
  props: Record<string, any>;
  children: VNode[];
  _component?: Vue;
  _domNode?: Node | null;
};

/** 创建一个根Vue组件 */
export function createApp(options: VueOptions) {
  const vue = new Vue(options);
  vue._isAppRoot = true;
  vue._innerSetName("App");
  // @ts-ignore
  window.__vue_app__ = vue;
  return vue;
}

/** Vue对象 */
export default class Vue {
  name?: string;
  _isAppRoot: boolean = false; // 是否是App根组件
  _options: VueOptions; // 选项
  el?: HTMLElement | null; // 父级element元素
  components: Record<string, VueOptions>; // 声明的组件
  _componentInstances: Record<string, Vue>; // 使用到的组件实例
  proxy: Record<string, any>; // 代理对象，用于访问this.data、this.methods、this.props
  watchers: Record<string, (newVal: any, oldVal: any) => void>; // 监听器
  _lastVNode?: VNode; // 上一次渲染的vnode
  mounted: () => void; // 挂载完成回调
  destroyed: () => void; // 销毁回调
  render: (h: (tag: string, opts: any, children?: any[]) => VNode) => VNode; // 渲染函数
  _updateQueued: boolean = false; // 是否有更新任务

  constructor(options: VueOptions) {
    this._options = options;
    // name
    this.name = options.name;
    // el
    this.el = options.el;
    // components
    this.components = options.components || {};
    this._componentInstances = {};
    // data
    this.proxy = this._proxyThis(
      options.data?.(),
      options.methods,
      options.props
    );
    // watch
    this.watchers = options.watch || {};
    // lifecycle
    this.mounted = options.mounted || (() => {});
    this.destroyed = options.destroyed || (() => {});

    this.render = options.render.bind(this.proxy);

    this._init();
  }

  _init() {
    this.mounted();
    // 首次渲染
    this._queueRender.call(this);
  }

  /** 设置组件name，如果已在options中设置，则忽略 */
  _innerSetName(name: string) {
    if (!this.name) this.name = name;
  }

  // 为当前Vue实例创建一个代理
  _proxyThis(
    data: Record<string, any> = {},
    methods: Record<string, any> = {},
    props: string[] = []
  ) {
    const that = this;
    const obj = { ...data, ...methods };
    for (const key of props) {
      obj[key] = undefined;
    }
    const proxy = new Proxy(obj, {
      get: (target, p: string, receiver) => {
        const value = target[p];
        if (typeof value === "function") {
          // 为methods绑定proxy
          return value.bind(proxy);
        }
        return value;
      },
      set: (target, p: string, newValue, receiver) => {
        const oldValue = target[p];
        target[p] = newValue;
        // 触发watcher
        if (p in that.watchers) {
          that.watchers[p].call(proxy, newValue, oldValue);
        }
        // 重新渲染
        that._queueRender.call(that);
        return true;
      },
    });
    return proxy;
  }

  // 异步更新队列，在微任务结束后执行渲染
  _queueRender() {
    // 防抖
    if (!this._updateQueued) {
      this._updateQueued = true;
      Promise.resolve().then(() => {
        this._doRender();
        this._updateQueued = false;
      });
    }
  }

  // 执行渲染
  _doRender() {
    if (this.el) {
      const vnodeTree = this.render(h.bind(this));
      // 根组件首次渲染，清空el下所有元素，防止VNode和真实dom不一致
      if (this._isAppRoot) {
        // 首次渲染
        if (!this._lastVNode) {
          this.el.innerHTML = "";
        }
      }

      // 使用diff算法重新渲染子树
      this._renderDomWithDiff(
        this._lastVNode ? [this._lastVNode] : [],
        [vnodeTree],
        this.el
      );
      // 记录下来
      this._lastVNode = vnodeTree;
    }
  }

  destroy() {
    // 销毁子组件
    this._unmountComponents();
    // 销毁自己
    this.destroyed();
  }

  /** 销毁特定的子组件实例 */
  _unmountComponent(key: string) {
    if (this._componentInstances.hasOwnProperty(key)) {
      this._componentInstances[key].destroy();
      delete this._componentInstances[key];
    }
  }

  /** 未销毁的子组件实例全部销毁 */
  _unmountComponents() {
    for (const key in this._componentInstances) {
      this._componentInstances[key].destroy();
    }
    this._componentInstances = {};
  }

  /**
   * 遍历以及diff一组新旧VNode，同时对parent下的真实dom进行创建、更新、移动或删除。
   * 从而以最小的变动让真实dom和最新VNode树保持一致。
   *
   * 原理：
   * 使用双端diff算法，新旧前后共4个指针从两端向中间移动并尝试相互匹配。
   * 匹配到key相同的VNode，复用dom，调用patch更新属性，然后移动或保持dom位置（根据指针）；
   * 未匹配到key相同的VNode，创建dom并插入到parent某个位置下（根据指针）；
   * 最后，如果旧的VNode有多余，删除parent下对应的dom。
   *
   * @param oldChildren 旧children
   * @param newChildren 新children
   * @param parent 父级dom元素
   */
  _renderDomWithDiff(
    oldChildren: VNode[],
    newChildren: VNode[],
    parent: HTMLElement
  ) {
    let oldStartIdx = 0;
    let oldEndIdx = oldChildren.length - 1;
    let newStartIdx = 0;
    let newEndIdx = newChildren.length - 1;

    while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
      // 旧节点可能已被匹配，被置空了，跳过
      if (!oldChildren[oldStartIdx]) {
        oldStartIdx++;
      } else if (!oldChildren[oldEndIdx]) {
        oldEndIdx--;
      } else if (
        oldChildren[oldStartIdx].key === newChildren[newStartIdx].key
      ) {
        // 旧开始节点和新开始节点key相同，无需移动dom
        console.log(
          `【${this.name}】[${oldChildren[oldStartIdx].key}]匹配到节点`
        );
        this._patch(oldChildren[oldStartIdx], newChildren[newStartIdx], parent);
        oldStartIdx++;
        newStartIdx++;
      } else if (oldChildren[oldEndIdx].key === newChildren[newEndIdx].key) {
        // 旧结束节点和新结束节点key相同，无需移动dom
        console.log(
          `【${this.name}】[${oldChildren[oldEndIdx].key}]匹配到节点`
        );
        this._patch(oldChildren[oldEndIdx], newChildren[newEndIdx], parent);
        oldEndIdx--;
        newEndIdx--;
      } else if (oldChildren[oldStartIdx].key === newChildren[newEndIdx].key) {
        // 旧开始节点和新结束节点key相同，需要移动到旧结束节点后面
        console.log(
          `【${this.name}】[${oldChildren[oldStartIdx].key}]匹配到节点`
        );
        const node = this._patch(
          oldChildren[oldStartIdx],
          newChildren[newEndIdx],
          parent
        );
        parent.insertBefore(node, oldChildren[oldEndIdx]._domNode!.nextSibling);
        oldStartIdx++;
        newEndIdx--;
      } else if (oldChildren[oldEndIdx].key === newChildren[newStartIdx].key) {
        // 旧结束节点和新开始节点key相同，需要移动到旧开始节点前面
        console.log(
          `【${this.name}】[${oldChildren[oldEndIdx].key}]匹配到节点`
        );
        const node = this._patch(
          oldChildren[oldEndIdx],
          newChildren[newStartIdx],
          parent
        );
        parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
        oldEndIdx--;
        newStartIdx++;
      } else {
        // 前面四种情况都不匹配，尝试从旧数组中匹配新开始节点
        const idxInOld = oldChildren.findIndex(
          (node) => node && node.key === newChildren[newStartIdx].key
        );
        if (idxInOld !== -1) {
          // 找到了，需要移动到旧开始节点前面
          console.log(
            `【${this.name}】[${newChildren[newStartIdx].key}]匹配到节点`
          );
          const node = this._patch(
            oldChildren[idxInOld],
            newChildren[newStartIdx],
            parent
          );
          parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
          // 置空，防止反复鞭尸
          // @ts-ignore
          oldChildren[idxInOld] = undefined;
        } else {
          // 找不到，创建dom，插入到旧开始节点前面
          console.log(
            `【${this.name}】[${newChildren[newStartIdx].key}]是新增节点`
          );
          // 插入到oldStartIdx前面正好保证顺序
          const node = this._patch(undefined, newChildren[newStartIdx], parent);
          parent.insertBefore(node, oldChildren[oldStartIdx]._domNode!);
        }
        newStartIdx++;
      }
    }

    // 新节点比较完毕，旧节点还有剩余
    if (oldStartIdx <= oldEndIdx) {
      for (let i = oldStartIdx; i <= oldEndIdx; i++) {
        if (!oldChildren[i]) continue;
        // 删除掉对应的dom
        console.log(`【${this.name}】[${oldChildren[i].key}]是删除节点`);
        parent.removeChild(oldChildren[i]._domNode!);
        // 如果是组件，需要销毁
        if (this.components.hasOwnProperty(oldChildren[i].type)) {
          this._unmountComponent(oldChildren[i].key);
        }
      }
    } else if (newStartIdx <= newEndIdx) {
      // 旧节点比较完毕，新节点还有剩余
      for (let i = newStartIdx; i <= newEndIdx; i++) {
        console.log(`【${this.name}】[${newChildren[i].key}]是新增节点`);
        // 创建dom，插入到parent末尾
        const node = this._patch(undefined, newChildren[i], parent);
        parent.insertBefore(node, newChildren[newEndIdx + 1]?._domNode || null);
      }
    }
  }

  /**
   * 创建或更新真实dom节点。
   *
   * 1. 如果有oldVNode：复用dom，只更新dom属性。
   * 2. 如果没有oldVNode：创建新的dom
   * 3. 递归处理children
   * @param oldVNode 旧的vnode
   * @param vnode 新的vnode
   * @param parent 父级dom元素
   * @returns 真实dom节点
   */
  _patch(oldVNode: VNode | undefined, vnode: VNode, parent: HTMLElement): Node {
    // 1. 文本节点
    if (vnode.type === "text") {
      if (!oldVNode) {
        // 没有则创建
        const node = document.createTextNode(vnode.props.textContent);
        vnode._domNode = node;
      } else {
        if (oldVNode.props.textContent !== vnode.props.textContent) {
          oldVNode._domNode!.textContent = vnode.props.textContent;
        }
        vnode._domNode = oldVNode._domNode;
      }
      return vnode._domNode!;
    }

    // 2. 组件节点
    if (this.components.hasOwnProperty(vnode.type)) {
      if (!oldVNode) {
        // 没有则创建
        let component = this._componentInstances[vnode.key];
        if (!component) {
          component = new Vue(this.components[vnode.type]);
          component._innerSetName(vnode.type);
          this._componentInstances[vnode.key] = component;
        }
        // 更新proxy的props
        for (const key in vnode.props) {
          component.proxy[key] = vnode.props[key];
        }
        vnode._component = component;
      } else {
        // 否则只需更新
        for (const key in vnode.props) {
          oldVNode._component!.proxy[key] = vnode.props[key];
        }
        vnode._component = oldVNode._component;
      }

      vnode._component!.el = parent;
      // 子组件去渲染自己的元素即可
      vnode._component!._doRender();
      vnode._domNode = vnode._component!._lastVNode!._domNode!;
      return vnode._domNode;
    }

    // 3. 其他节点
    if (!oldVNode) {
      // 没有则创建
      const element = document.createElement(vnode.type);
      // 设置属性
      _updateProps(element, vnode.props);
      vnode._domNode = element;
    } else {
      // 否则只需更新
      _updateProps(
        oldVNode._domNode as HTMLElement,
        vnode.props,
        oldVNode.props
      );
      vnode._domNode = oldVNode._domNode;
    }

    // 递归处理children
    this._renderDomWithDiff(
      oldVNode?.children || [],
      vnode.children,
      vnode._domNode as HTMLElement
    );

    return vnode._domNode!;
  }
}

/** 差量更新dom节点的属性 */
function _updateProps(
  element: HTMLElement,
  props: Record<string, any>,
  oldProps?: Record<string, any>
) {
  for (const key in props) {
    if (key === "style") {
      // style
      if (oldProps && oldProps["style"]) {
        // 清空旧的style
        for (const styleKey in oldProps["style"]) {
          // @ts-ignore
          element.style[styleKey] = "";
        }
      }
      Object.assign(element.style, props[key]);
    } else if (key.startsWith("on")) {
      // 事件
      if (oldProps && oldProps[key]) {
        // 移除旧的事件
        element.removeEventListener(key.slice(2).toLowerCase(), oldProps[key]);
      }
      element.addEventListener(key.slice(2).toLowerCase(), props[key]);
    } else {
      // 其他
      if (oldProps && oldProps[key]) {
        // 移除旧属性
        element.removeAttribute(key);
      }
      element.setAttribute(key, props[key]);
    }
  }
}

/** h函数帮助生成虚拟节点VNode */
function h(
  type: string,
  props: any,
  children: (VNode | string | null)[] = []
): VNode {
  // 存储为VNode
  let vnode: VNode = {
    type,
    props,
    children: [],
    key: _createKey(type, props.key),
  };
  let nonNullChildren = children.filter((child) => child !== null);
  // 添加children
  for (let idx = 0; idx < nonNullChildren.length; idx++) {
    const child = nonNullChildren[idx];
    if (typeof child === "string") {
      vnode.children.push({
        type: "text",
        props: { textContent: child },
        children: [],
        key: _createKey("text", undefined, idx),
      });
    } else {
      // 重新计算key
      child.key = _createKey(child.type, child.props.key, idx);
      vnode.children.push(child);
    }
  }

  return vnode;
}

/**
 * 创建VNode的key
 * 默认key为type-idx
 * 如果提供了props.key，则为type-key
 */
function _createKey(
  type: string,
  key: string | undefined,
  idx: number = 0
): string {
  return key ? `${type}-${key}` : `${type}-${idx}`;
}
